Blender에서 GLB 파일 최적화하기

배경

웹에서 Three.js 적용한 화면 로딩시 GPU사용량이 과함. 해당 사용량을 줄이기 위한 GLB 파일의 용량과 성능 최적화가 필요한 상황. Blender에서 폴리곤과 정점을 줄이면서도 품질을 유지하는 방법을 찾고 있었음.

GLB 파일 최적화 핵심 개념

3D 모델은 3각형의 폴리곤으로 구성됨, 하나의 폴리곤은 3개의 점(정점) 사이의 면임.
최적화를 하지 않으면 점이 같은 위치에 있더라도 여러개의 점인 것 처럼 만듦. GLB 모델 단에서 이 정점을 최적화 해야 효율적으로 최적화 가능함.

즉, 모델링 변경 시 폴리곤만 줄이면 안됨. 정점도 같이 줄여야 최적화됨
다만 여러면이 모이는 정점은 면마다 normalize를 위해서 줄이면 안됨
줄이는 정점은 평면에서 필요없는 정점

// Three.js 구조
const vertices = new Float32Array([
  0, 0, 0,  // 정점 0
  1, 0, 0,  // 정점 1
  0, 1, 0,  // 정점 2
  1, 1, 0   // 정점 3
]);

const indices = new Uint16Array([
  0, 1, 2,  // 폴리곤 1
  1, 3, 2   // 폴리곤 2
]);

부드러운 표면의 원리

부드러워 보이는 건 정점이 많아서가 아니라 Normal 벡터로 만드는 시각적 효과:

  • Flat Shading: 각 면마다 수직 Normal → 각져 보임
  • Smooth Shading: 주변 면들의 평균 Normal → 부드럽게 보임

같은 폴리곤 수라도 Normal 값만 바꾸면 부드러워 보임. 이 Normal 데이터는 GLB에 완전히 포함되어 Three.js에서 그대로 작동함.

Blender 최적화 3단계

1. 폴리곤 감소 (Decimate)

  • 삼각형 개수 감소
  • 인덱스 데이터 감소

2. 정점 병합 (Merge by Distance)

  • 중복/불필요한 정점 제거(면과 면이 만나는 정점은 normalize를 위해서 병합하면 안됨)
  • geometry.attributes.position 크기 감소
  • 같은 위치의 정점들을 하나로 병합

3. Limited Dissolve (평면 정리)

  • 평평한 면에서 불필요한 중간 정점/엣지 제거
  • 사각형 면인데 폴리곤이 여러 개인 경우 획기적으로 감소
  • 가장 효과적인 최적화 방법
  • 중간점이 있으면 구부러지거나 복잡한 텍스처에서 장점이 있음.(적당한 제거만 필요)

미사용 정점: 아무 면(face)에도 연결되지 않은 고립된 정점. 삭제해도 모델 형태에 영향 없음.(AI로 만든 모델은 평평한 면에도 여러개의 정점을 부여하는데 사용하기 전에 사용되지 않는 정점은 정리가 필요)

Blender Python 범용 최적화 스크립트

한번에 스크립팅으로 처리하기 위해

UI에서 수동으로 선택하는 대신 오브젝트 이름으로 타겟팅 할 수 있다. 반복적인 작업이나 특정 대상을 드래그 하는 대신 이름을 부여하고 스크립트로 해결하자.

import bpy
import math
import re

# ============================================================
# 최적화 설정
# ============================================================

# 폴리곤 감소 비율 (0.0 ~ 1.0)
DECIMATE_RATIO = 0.95

# 평면 판정 각도 (degree)
PLANAR_ANGLE = 1

# 정점 병합 임계값 (meter)
MERGE_DISTANCE = 0.0001

# 타겟팅 모드 선택
# 'SELECTED' | 'ALL' | 'VISIBLE' | 'NAMES' | 'PREFIX' | 'CONTAINS' | 'REGEX'
TARGET_MODE = 'SELECTED'

# 타겟팅 설정 (모드에 따라 사용)
TARGET_NAMES = ['Body', 'Head', 'Arm_L', 'Arm_R']
NAME_PREFIX = 'Character_'
NAME_CONTAINS = '_optimized'
REGEX_PATTERN = r'^Character_(Body|Head|Arm_[LR])$'
EXCLUDE_NAMES = ['Camera', 'Light', 'Ground']

# 최적화 옵션
USE_DECIMATE = False      # 단순 형상엔 비추천
USE_DISSOLVE = True       # 평면 정리 (핵심!)
USE_MERGE = True          # 중복 정점 제거
USE_CLEANUP = True        # 미사용 정점 제거
VERBOSE = True

# 최적화 필터 (너무 단순한 형상 제외)
MIN_VERTICES = 100        # 이보다 적으면 스킵
SKIP_SIMPLE_SHAPES = True # 이미 최적화된 형상 스킵

# 이름 변경 설정
RENAME_AFTER_OPTIMIZE = False
RENAME_SUFFIX = '_opt'    # 최적화 후 이름에 추가

# Export 설정
AUTO_EXPORT = False
EXPORT_PATH = '//optimized_model.glb'
EXPORT_SELECTED_ONLY = True
USE_DRACO_COMPRESSION = True

# ============================================================
# 타겟팅 함수
# ============================================================

def get_target_objects():
    """설정에 따라 타겟 오브젝트 반환"""
    objects = []
    
    if TARGET_MODE == 'SELECTED':
        objects = [obj for obj in bpy.context.selected_objects if obj.type == 'MESH']
        
    elif TARGET_MODE == 'ALL':
        objects = [obj for obj in bpy.data.objects if obj.type == 'MESH']
        
    elif TARGET_MODE == 'VISIBLE':
        objects = [obj for obj in bpy.data.objects 
                  if obj.type == 'MESH' and obj.visible_get()]
        
    elif TARGET_MODE == 'NAMES':
        for name in TARGET_NAMES:
            if name in bpy.data.objects:
                obj = bpy.data.objects[name]
                if obj.type == 'MESH':
                    objects.append(obj)
            else:
                print(f"⚠️ '{name}' 오브젝트를 찾을 수 없습니다")
                
    elif TARGET_MODE == 'PREFIX':
        for obj in bpy.data.objects:
            if obj.type == 'MESH' and obj.name.startswith(NAME_PREFIX):
                objects.append(obj)
                
    elif TARGET_MODE == 'CONTAINS':
        for obj in bpy.data.objects:
            if obj.type == 'MESH' and NAME_CONTAINS in obj.name:
                objects.append(obj)
                
    elif TARGET_MODE == 'REGEX':
        pattern = re.compile(REGEX_PATTERN)
        for obj in bpy.data.objects:
            if obj.type == 'MESH' and pattern.match(obj.name):
                objects.append(obj)
    
    # 제외 목록 필터링
    objects = [obj for obj in objects if obj.name not in EXCLUDE_NAMES]
    
    return objects

# ============================================================
# 최적화 필터
# ============================================================

def should_optimize(obj):
    """최적화가 필요한지 판단"""
    if not SKIP_SIMPLE_SHAPES:
        return True
    
    stats = get_mesh_stats(obj)
    
    # 정점이 너무 적으면 스킵
    if stats['vertices'] < MIN_VERTICES:
        return False
    
    # 이미 최적화된 형상 (정점 대비 면이 적절)
    if stats['faces'] / stats['vertices'] < 1.5:
        return False
    
    return True

# ============================================================
# 메시 정보 및 최적화
# ============================================================

def get_mesh_stats(obj):
    """메시 통계 반환"""
    return {
        'vertices': len(obj.data.vertices),
        'edges': len(obj.data.edges),
        'faces': len(obj.data.polygons)
    }

def optimize_mesh(obj):
    """단일 메시 최적화"""
    stats_before = get_mesh_stats(obj)
    
    # 오브젝트 선택
    bpy.ops.object.select_all(action='DESELECT')
    obj.select_set(True)
    bpy.context.view_layer.objects.active = obj
    
    # Edit Mode 작업
    bpy.ops.object.mode_set(mode='EDIT')
    bpy.ops.mesh.select_all(action='SELECT')
    
    # 1. 정점 병합 (중복 제거)
    if USE_MERGE:
        bpy.ops.mesh.remove_doubles(threshold=MERGE_DISTANCE)
    
    # 2. Limited Dissolve (평면 정리) - 핵심!
    if USE_DISSOLVE:
        angle_radians = math.radians(PLANAR_ANGLE)
        bpy.ops.mesh.dissolve_limited(
            angle_limit=angle_radians,
            use_dissolve_boundaries=False,
            delimit={'NORMAL'}
        )
    
    # 3. 미사용 데이터 제거
    if USE_CLEANUP:
        bpy.ops.mesh.delete_loose()
    
    bpy.ops.object.mode_set(mode='OBJECT')
    
    # 4. Decimate (폴리곤 감소) - 선택적
    if USE_DECIMATE and DECIMATE_RATIO < 1.0:
        decimate = obj.modifiers.new(name="Decimate_Optimize", type='DECIMATE')
        decimate.ratio = DECIMATE_RATIO
        bpy.ops.object.modifier_apply(modifier="Decimate_Optimize")
    
    stats_after = get_mesh_stats(obj)
    return stats_before, stats_after

def rename_object(obj):
    """최적화 후 이름 변경"""
    if RENAME_AFTER_OPTIMIZE:
        new_name = obj.name + RENAME_SUFFIX
        obj.name = new_name
        return new_name
    return obj.name

# ============================================================
# 출력 및 Export
# ============================================================

def print_stats(name, before, after):
    """통계 출력"""
    if before['vertices'] == 0:
        return
    
    vert_reduction = ((before['vertices'] - after['vertices']) / before['vertices'] * 100)
    face_reduction = ((before['faces'] - after['faces']) / before['faces'] * 100)
    
    print(f"\n[{name}]")
    print(f"  정점: {before['vertices']:,} → {after['vertices']:,} ({vert_reduction:.1f}% 감소)")
    print(f"  면:   {before['faces']:,} → {after['faces']:,} ({face_reduction:.1f}% 감소)")

def export_glb():
    """GLB 파일로 Export"""
    if not AUTO_EXPORT:
        return
    
    try:
        bpy.ops.export_scene.gltf(
            filepath=bpy.path.abspath(EXPORT_PATH),
            export_format='GLB',
            use_selection=EXPORT_SELECTED_ONLY,
            export_apply=True,
            export_hierarchy=True,
            export_draco_mesh_compression_enable=USE_DRACO_COMPRESSION,
            export_draco_mesh_compression_level=6
        )
        print(f"\n✓ GLB 파일 Export 완료: {EXPORT_PATH}")
    except Exception as e:
        print(f"\n❌ Export 실패: {e}")

# ============================================================
# 메인 실행
# ============================================================

def main():
    """메인 실행 함수"""
    print("="*60)
    print("Blender 메시 최적화 스크립트 v2.0")
    print("="*60)
    print(f"\n타겟팅 모드: {TARGET_MODE}")
    
    # 설정 정보 출력
    if TARGET_MODE == 'NAMES':
        print(f"타겟 오브젝트: {TARGET_NAMES}")
    elif TARGET_MODE == 'PREFIX':
        print(f"접두사: '{NAME_PREFIX}'")
    elif TARGET_MODE == 'CONTAINS':
        print(f"포함 문자열: '{NAME_CONTAINS}'")
    elif TARGET_MODE == 'REGEX':
        print(f"정규식: '{REGEX_PATTERN}'")
    
    if EXCLUDE_NAMES:
        print(f"제외 목록: {EXCLUDE_NAMES}")
    
    print("\n최적화 옵션:")
    print(f"  - Decimate: {'ON' if USE_DECIMATE else 'OFF'} ({DECIMATE_RATIO*100:.0f}%)")
    print(f"  - Dissolve: {'ON' if USE_DISSOLVE else 'OFF'} ({PLANAR_ANGLE}°)")
    print(f"  - Merge: {'ON' if USE_MERGE else 'OFF'} ({MERGE_DISTANCE}m)")
    print(f"  - Cleanup: {'ON' if USE_CLEANUP else 'OFF'}")
    
    if SKIP_SIMPLE_SHAPES:
        print(f"  - 단순 형상 스킵: ON (최소 정점: {MIN_VERTICES})")
    
    print("="*60)
    
    # 타겟 오브젝트 가져오기
    target_objects = get_target_objects()
    
    if not target_objects:
        print("\n❌ 최적화할 메시 오브젝트가 없습니다!")
        print("\n현재 씬의 메시 오브젝트:")
        for obj in bpy.data.objects:
            if obj.type == 'MESH':
                print(f"  - {obj.name}")
        return
    
    print(f"\n✓ {len(target_objects)}개 오브젝트 발견:")
    for obj in target_objects:
        stats = get_mesh_stats(obj)
        status = "최적화 예정" if should_optimize(obj) else "스킵 (단순 형상)"
        print(f"  - {obj.name} ({stats['vertices']} 정점) → {status}")
    print()
    
    # 최적화 실행
    total_before = {'vertices': 0, 'edges': 0, 'faces': 0}
    total_after = {'vertices': 0, 'edges': 0, 'faces': 0}
    optimized_count = 0
    skipped_count = 0
    
    for i, obj in enumerate(target_objects, 1):
        print(f"\n[{i}/{len(target_objects)}] {obj.name}")
        
        if not should_optimize(obj):
            print("  ⊘ 스킵 - 이미 최적화된 형상")
            skipped_count += 1
            continue
        
        try:
            stats_before, stats_after = optimize_mesh(obj)
            
            for key in total_before.keys():
                total_before[key] += stats_before[key]
                total_after[key] += stats_after[key]
            
            if VERBOSE:
                print_stats(obj.name, stats_before, stats_after)
            
            # 이름 변경
            new_name = rename_object(obj)
            if new_name != obj.name:
                print(f"  → 이름 변경: {new_name}")
            
            optimized_count += 1
            
        except Exception as e:
            print(f"  ❌ 오류 발생: {e}")
            continue
    
    # 최종 통계
    print("\n" + "="*60)
    print("최적화 완료!")
    print("="*60)
    print(f"처리: {optimized_count}개 최적화, {skipped_count}개 스킵")
    
    if optimized_count > 0:
        print_stats("전체 통계", total_before, total_after)
    
    # Export
    export_glb()
    print("\n")

if __name__ == "__main__":
    main()

정규식 패턴 매칭

import re

def get_target_objects_regex():
    """정규식으로 복잡한 패턴 매칭"""
    pattern = r'^Character_(Body|Head|Arm_[LR])$'
    objects = []
    
    for obj in bpy.data.objects:
        if obj.type == 'MESH' and re.match(pattern, obj.name):
            objects.append(obj)
    
    return objects

패턴 예시:

  • ^Character_.*$: Character_로 시작하는 모든 오브젝트
  • .*_(LOD0|LOD1)$: LOD0 또는 LOD1로 끝나는 오브젝트
  • ^(Body|Head)_.*: Body 또는 Head로 시작하는 오브젝트3
