Reverse Proxy Nginx

개요

nginX를 이용해서 reverse Proxy를 구축하자.

reverse Proxy

서버 앞단에 위치해서 외부요청을 내부 서버로 전달하는 역할을 하는 프록시

NginX에서 요청 url을 통해 구분해서 백엔드로 보내는 역할을 수행, 직접 백엔드에 접근하지 않고 NginX가 대신 처리

  • 보안 : backend URL을 숨길 수 있음.
  • 로드밸런싱 : 여러 백엔드로 트래픽 분산
  • SSL 종료 : HTTPS 처리를 NGINX에서만 동작, 내부는 HTTP를 통해 진행
  • 캐싱: 자주 쓰는 응답은 캐싱처리

단일 백엔드 서버 프록시

location 을 통해 url을 구분하고 주소를 

server {
    listen 80;
    server_name example.com;

    location / {
        root /usr/share/nginx/html;
        index index.html;
    }

    # /api 로 들어오는 요청은 백엔드 서버로 전달
    location /api/ {
        proxy_pass http://localhost:8080/;   # 백엔드 서버 주소
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}

로드밸런싱

http {
    # backend 서버 둘중 하나를 선택해서 전송작업
    upstream backend {
        server 127.0.0.1:8080;
        server 127.0.0.1:8081;
    }

    server {
        listen 80;

        location /api/ {
            proxy_pass http://backend/;
        }
    }
}

SSL종료

백과 프론트의 SSL 제외로 내부 ssl_certificate, ssl_protocols, cipher_suites 설정을 안해도 상관없음.

server {
    listen 443 ssl;
    server_name example.com;

    ssl_certificate /etc/nginx/ssl/example.crt;
    ssl_certificate_key /etc/nginx/ssl/example.key;

    location /api/ {
        proxy_pass http://127.0.0.1:8080/;  # 내부는 HTTP
    }
}

1. 기본 구조

# 1) 백엔드 서버 그룹 정의 (로드밸런싱 가능)
upstream backend {
    server 127.0.0.1:8080;    # WAS 1
    # server 127.0.0.1:8081; # WAS 2 (필요 시 추가)
    keepalive 32;
}

# 2) HTTP → HTTPS 리다이렉트
server {
    listen 80;
    server_name example.com;
    return 301 https://$host$request_uri;
}

# 3) HTTPS 서버 블록
server {
    listen 443 ssl http2;
    server_name example.com;

    # SSL 인증서
    ssl_certificate     /etc/nginx/ssl/example.crt;
    ssl_certificate_key /etc/nginx/ssl/example.key;
    ssl_protocols       TLSv1.2 TLSv1.3;
    ssl_ciphers         HIGH:!aNULL:!MD5;

    # 공통 설정
    client_max_body_size 20M;   # 업로드 제한
    proxy_read_timeout   60s;   # 백엔드 응답 대기 시간

    # 정적 파일 (프론트엔드 빌드 파일)
    location / {
        root /usr/share/nginx/html;
        index index.html;
        try_files $uri /index.html;
    }

    # API 요청은 백엔드로 전달
    location /api/ {
        proxy_pass http://backend/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    # 정적 파일 캐싱 (이미지, JS, CSS 등)
    location ~* \.(?:ico|css|js|gif|jpe?g|png|woff2?|eot|ttf|svg|mp4)$ {
        expires 7d;
        access_log off;
        add_header Cache-Control "public";
        root /usr/share/nginx/html;
    }
}

2. 주요 포인트

1) upstream 블록

  • 내부 백엔드 서버를 정의하고, 필요하면 여러 개 등록해 로드밸런싱 가능
  • keepalive로 커넥션 재사용해 성능 향상

2) HTTP → HTTPS 리다이렉트

  • 모든 사용자가 무조건 HTTPS로 접속하도록 강제

3) SSL 최적화

  • TLSv1.2, TLSv1.3만 허용
  • 불필요한 약한 암호화 알고리즘 차단

4) /api/ Reverse Proxy

  • API 요청을 백엔드로 전달
  • WebSocket도 지원하도록 Upgrade/Connection 헤더 설정

5) 정적 파일 최적화

  • React, Vue, Next.js 같은 SPA는 try_files $uri /index.html; 로 라우팅 지원
  • 이미지, JS, CSS는 캐싱 헤더 추가

3. 실무에서 자주 추가하는 옵션

  • Rate Limit (DDoS 방어)
limit_req_zone $binary_remote_addr zone=api_limit:10m rate=10r/s;

location /api/ {
    limit_req zone=api_limit burst=20 nodelay;
    proxy_pass http://backend/;
}
  • 보안 헤더
add_header X-Frame-Options "SAMEORIGIN" always;
add_header X-Content-Type-Options "nosniff" always;
add_header Referrer-Policy "strict-origin-when-cross-origin" always;
  • 압축 전송
gzip on;
gzip_types text/plain text/css application/json application/javascript application/xml image/svg+xml;
gzip_min_length 256;

SPA + Spring boot 설정

# ====== 0) WebSocket 업그레이드 헤더 매핑 ======
map $http_upgrade $connection_upgrade {
    default     keep-alive;
    "websocket" upgrade;
}

# ====== 1) 백엔드 업스트림 (로드밸런싱 가능) ======
upstream spring_api {
    server 127.0.0.1:8080;   # Spring Boot
    # server 127.0.0.1:8081; # 필요 시 추가
    keepalive 32;
}

# ====== 2) HTTP -> HTTPS 리다이렉트 ======
server {
    listen 80;
    server_name example.com;
    return 301 https://$host$request_uri;
}

# ====== 3) HTTPS 서버 ======
server {
    listen 443 ssl http2;
    server_name example.com;

    # --- SSL ---
    ssl_certificate     /etc/nginx/ssl/example.crt;
    ssl_certificate_key /etc/nginx/ssl/example.key;
    ssl_protocols       TLSv1.2 TLSv1.3;
    ssl_ciphers         HIGH:!aNULL:!MD5;

    # --- 공통 ---
    client_max_body_size 20m;
    proxy_read_timeout   60s;
    proxy_send_timeout   60s;

    # --- 보안 헤더(권장) ---
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Permissions-Policy "geolocation=(), microphone=(), camera=()" always;

    # --- 압축(권장) ---
    gzip on;
    gzip_min_length 256;
    gzip_comp_level 5;
    gzip_types
        text/plain text/css application/json application/javascript
        application/xml application/rss+xml text/javascript image/svg+xml;

    # ====== 3-1) 정적 프론트 (SPA) ======
    # 빌드 산출물 경로 예: /var/www/app  (React build / Next.js static export)
    root /var/www/app;
    index index.html;

    # 해시된 정적 자원은 오래 캐시
    location ~* \.(?:css|js|mjs|ico|gif|jpe?g|png|webp|avif|woff2?|eot|ttf|svg)$ {
        access_log off;
        expires 30d;
        add_header Cache-Control "public, max-age=2592000, immutable";
        try_files $uri =404;
    }

    # SPA 라우팅 (브라우저 새로고침 시 404 방지)
    location / {
        try_files $uri /index.html;
    }

    # ====== 3-2) API 리버스 프록시 (Spring Boot) ======
    location /api/ {
        proxy_pass http://spring_api/;
        proxy_http_version 1.1;

        # 원본 헤더 전달
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # WebSocket 지원
        proxy_set_header Upgrade    $http_upgrade;
        proxy_set_header Connection $connection_upgrade;

        # 필요 시 요청/응답 헤더 조정
        # proxy_set_header Authorization $http_authorization;
        # proxy_hide_header X-Powered-By;
    }

    # ====== 3-3) 헬스체크/상태 페이지 ======
    location = /healthz {
        access_log off;
        add_header Content-Type text/plain;
        return 200 'ok';
    }

    # ====== 3-4) (옵션) API 레이트 리밋 ======
